%% Mathematics

\newcommand{\Runderline}[1]{%
\begin{tikzpicture}[anchor=base,baseline]
    \node[inner sep=0,outer sep=0] (N) at (0,0) {#1};
    \coordinate (L) at ($(N.base west)+(0.1ex,-0.3ex)$);
    \coordinate (R) at ($(N.base east)+(-0.1ex,-0.3ex)$);
    \draw (L) -- (R);
\end{tikzpicture}}

\newcommand{\Rdoubleunderline}[1]{%
\begin{tikzpicture}[anchor=base,baseline]
    \node[inner sep=0,outer sep=0] (N) at (0,0) {#1};
    \coordinate (L) at ($(N.base west)+(0.1ex,-0.3ex)$);
    \coordinate (R) at ($(N.base east)+(-0.1ex,-0.3ex)$);
    \draw (L) -- (R);
    \draw ($(L)+(0,-0.3ex)$) -- ($(R)+(0,-0.3ex)$);
\end{tikzpicture}
}

\newcommand{\Runderarrow}[1]{%
\begin{tikzpicture}[anchor=base,baseline]
    \node[inner sep=0,outer sep=0] (N) at (0,0) {#1};
    \coordinate (L) at ($(N.base west)+(0.1ex,-0.3ex)$);
    \coordinate (R) at ($(N.base east)+(-0.1ex,-0.3ex)$);
    \draw (L) -- (R) -- ++(-0.3ex,-0.3ex);
\end{tikzpicture}
}
\def\RMstyleD{\displaystyle}
\def\RMstyleT{\textstyle}
\def\RMstyleS{\scriptstyle}
\def\RMstyles{\scriptscriptstyle}
\def\RMsavedstyle{\csname RMstyle\RMstylecase\endcsname}

\newcommand{\RMthisstyle}[1]%
{%
    \mathchoice%
    {\edef\RMstylecase{D}#1}%
    {\edef\RMstylecase{T}#1}%
    {\edef\RMstylecase{S}#1}%
    {\edef\RMstylecase{s}#1}%
}

\newcommand{\Rvec}[1]%
{\RMthisstyle{\Runderline{\ensuremath{\RMsavedstyle#1}}}}
\newcommand{\Rmat}[1]%
{\RMthisstyle{\Rdoubleunderline{\ensuremath{\RMsavedstyle#1}}}}
\newcommand{\Rdir}[1]%
{\RMthisstyle{\Runderarrow{\ensuremath{\RMsavedstyle#1}}}}

\newcommand{\Rdiff}[3][]{%
    \ifthenelse{\equal{#1}{}}%
        {\frac{d #2}{d #3}}%
        {\frac{d^{#1} #2}{d #3^{#1}}}%
}

\newcommand{\Rpdiff}[3][]{%
    \ifthenelse{\equal{#1}{}}%
        {\frac{\partial #2}{\partial #3}}%
        {\frac{\partial^{#1} #2}{\partial #3^{#1}}}%
}

\newcommand{\RDiff}[3][]{%
    \ifthenelse{\equal{#1}{}}%
        {\frac{D #2}{D #3}}%
        {\frac{D^{#1} #2}{D #3^{#1}}}%
}
